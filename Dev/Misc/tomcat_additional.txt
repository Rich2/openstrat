11. Install snap
  sudo apt install snapd
    Install certbot
  sudo snap install --classic certbot
  >>certbot 5.1.0 from Certbot Project (certbot-eff✓) installed
  sudo ln -s /snap/bin/certbot /usr/bin/certbot

    Install certificate
  sudo systemctl stop <ENTER TOMCAT SERVICE>
  sudo certbot certonly --standalone
  <ENTER EMAIL>
  <AGREE OR DISAGREE *2>
  <ENTER DOMAIN NAME>

    Configure permissions to certs..
  sudo chgrp -R <ENTER TOMCAT USER> /etc/letsencrypt/live/
  sudo chgrp -R <ENTER TOMCAT USER> /etc/letsencrypt/archive/
  sudo chmod -R 750 /etc/letsencrypt/live/
  sudo chmod -R 750 /etc/letsencrypt/archive/
  sudo chmod 640 /etc/letsencrypt/live/theoppositeofzen.com/privkey.pem
  sudo chmod 644 /etc/letsencrypt/live/theoppositeofzen.com/cert.pem
  sudo chmod 644 /etc/letsencrypt/live/theoppositeofzen.com/chain.pem
    Check permissions - if you dont have access then something wrong...
  ls -la /etc/letsencrypt/live/theoppositeofzen.com/

    Configure Tomcat to use 443 & link to ssl cert above
  nano /opt/tomcat/Base/conf/server.xml
  <Connector port="443" protocol="org.apache.coyote.http11.Http11NioProtocol"
         maxThreads="150" SSLEnabled="true" secure="true" scheme="https">
        <UpgradeProtocol className="org.apache.coyote.http2.Http2Protocol" />
        <SSLHostConfig>
            <Certificate certificateFile="/etc/letsencrypt/live/<ENTER DOMAIN NAME>/cert.pem"
                         certificateKeyFile="/etc/letsencrypt/live/<ENTER DOMAIN NAME>/privkey.pem"
                         certificateChainFile="/etc/letsencrypt/live/<ENTER DOMAIN NAME>/chain.pem" />
        </SSLHostConfig>
    </Connector>

    Start tomcat
  sudo systemctl start <ENTER TOMCAT SERVICE> 
  sudo systemctl status <ENTER TOMCAT SERVICE>
